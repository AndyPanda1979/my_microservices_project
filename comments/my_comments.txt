Docker
=================================================
docker ps           посмотреть контейнеры
docker ps -a        посмотреть все контейнеры (включая те, что уже отработали)
docker image ls     посмотреть список образов

если нужна справка по командам - docker команда --help (например: docker run --help)

docker run --name mongoDB --rm -d -p 27017:27017 mongo
+++++++
--name      задать имя контейнера
-d          запуск в фоне
--rm        удалить контейнер сразу после остановки
-p          проброс портов
